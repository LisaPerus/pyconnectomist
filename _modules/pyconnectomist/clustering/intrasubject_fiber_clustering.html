# System import
import os

# Clindmri import
from pyconnectomist.wrappers import ConnectomistWrapper
from pyconnectomist import DEFAULT_CONNECTOMIST_PATH

# Sets for checking arguments that can take values in a finite set
SUBJECT_TYPES = frozenset(["adult", "child"])
CENTROIDS = frozenset(["mean_fiber", "closest_fiber"])



[docs]def intrasubject_fiber_clustering(
        outdir,
        bundle_map_paths,
        unbiased_t1_path,
        voronoi_mask_path,
        dw_to_t1_trf_path,
        t1_to_tal_trf_path,
        t2_path,
        subject_id=None,
        subject_type="adult",
        discard_corpus_callosum_fibers=False,
        segment_right_hemisphere=True,
        segment_left_hemisphere=True,
        segment_interhemispheric=True,
        segment_cerebellum=True,
        roi_mask_path=None,
        roi_labels=None,
        roi_names=None,
        min_fiber_length=20,
        max_fiber_length=200,
        voxel_sampler_point_count=13,
        percent_of_bundle_extraction_for_first_clustering=60,
        perform_a_second_clustering=True,
        percent_of_bundle_extraction_for_second_clustering=40,
        max_distance_for_fasicle_merging=10,
        length_groups='20 35 50 65 80 95 110 130 150 175 200',
        nb_fibers_per_voxels=5,
        parcel_size=3,
        min_parcel_size=12,
        min_cluster_size=300,
        average_nb_of_cluster=40,
        average_cluster_size=2000,
        small_size_for_isolated_cluster=4,
        connectivity_matrix_threshold=1,
        min_nb_of_fibers_by_bundle_before_watershed=5,
        nb_of_fibers_per_voxel=1,
        min_nb_of_fibers_by_bundle_after_watershed=5,
        min_nb_of_fibers_by_bundle_in_output_bundle_file=10,
        centroid="mean_fiber",
        remove_temporary_files=True,
        path_connectomist=DEFAULT_CONNECTOMIST_PATH):
    """
    Wrapper to Connectomist's "Fiber clustering" tab.

    Parameters
    ----------
    outdir: str
        Path to Connectomist output work directory.
    bundle_map_paths: list of str
        Paths to the bundle maps.
    unbiased_t1_path: str
    voronoi_mask_path: str
    dw_to_t1_trf_path: str
    t1_to_tal_trf_path: str
    t2_path: str
    subject_id: str, optional
    subject_type: {"adult", "child"}, optional
    discard_corpus_callosum_fibers: bool, optional
    segment_right_hemisphere: bool, optional
    segment_left_hemisphere: bool, optional
    segment_interhemispheric: bool, optional
    segment_cerebellum: bool, optional
    roi_mask_path: None or str, optional
    roi_labels: None or list of str, optional
    roi_names: None or list of str, optional
    min_fiber_length: int, optional
    max_fiber_length: int, optional
    voxel_sampler_point_count: int, optional
    percent_of_bundle_extraction_for_first_clustering: int, optional
    perform_a_second_clustering: bool, optional
    percent_of_bundle_extraction_for_second_clustering: int, optional
    max_distance_for_fasicle_merging: int, optional
    length_groups: str, optional
    nb_fibers_per_voxels: int, optional
    parcel_size: int, optional
    min_parcel_size: int, optional
    min_cluster_size: int, optional
    average_nb_of_cluster: int, optional
    average_cluster_size: int, optional
    small_size_for_isolated_cluster: int, optional
    connectivity_matrix_threshold: int, optional
    min_nb_of_fibers_by_bundle_before_watershed: int, optional
    nb_of_fibers_per_voxel: int, optional
    min_nb_of_fibers_by_bundle_after_watershed: int, optional
    min_nb_of_fibers_by_bundle_in_output_bundle_file: int, optional
    centroid: {"mean_fiber", "closest_fiber"}
    remove_temporary_files: bool, optional
    path_connectomist: str, optional
        Path to the Connectomist executable.

    Returns
    -------
    outdir: str
        path to Connectomist's output directory.
    """

    required_files = [unbiased_t1_path, voronoi_mask_path, dw_to_t1_trf_path,
                      t1_to_tal_trf_path, t2_path] + bundle_map_paths

    if roi_mask_path is not None:
        required_files.append(roi_mask_path)

        # Check that 'roi_labels' and 'roi_names' are passed and are lists.
        if not (isinstance(roi_labels, list) and isinstance(roi_names, list)):
            raise ValueError("When the argument 'roi_mask_path' is passed "
                             "you have to pass the two lists: 'roi_labels' "
                             "and 'roi_names'.")

        if len(roi_labels) != len(roi_names):
            raise ValueError("'roi_labels' and 'roi_names' have to have the "
                             "same length.")

        roi_labels_and_names = [[x, y] for x, y in zip(roi_labels, roi_names)]

    else:
        roi_labels_and_names = []

    for path in required_files:
        if not os.path.isfile(path):
            ValueError("File does not exist: {}".format(path))

    if subject_type not in SUBJECT_TYPES:
        raise ValueError("Bad argument 'subject_type': {}, should be in {}"
                         .format(subject_type, SUBJECT_TYPES))

    if centroid not in CENTROIDS:
        raise ValueError("Bad argument 'centroid': {}, should be in {}"
                         .format(centroid, CENTROIDS))

    # Dict with all parameters for connectomist
    algorithm = "DWI-IntraSubject-Fiber-Clustering"
    parameters_dict = {

        # Options are ordered like in the GUI

        # ---------------------------------------------------------------------
        # "Input data" panel

        'fileNameFibers':                " ".join(bundle_map_paths),
        'fileNameUnbiasedT1':            unbiased_t1_path,
        'fileNameVoronoiMask':           voronoi_mask_path,
        'fileNameTransformationDwToT1':  dw_to_t1_trf_path,
        'fileNameTransformationT1ToTal': t1_to_tal_trf_path,
        'fileNameT2Picture':             t2_path,

        'subjectTypeAdult':      1 if subject_type == "adult" else 0,
        'subjectTypeChild':      1 if subject_type == "child" else 0,
        'discardCorpusCallosum': 2 if discard_corpus_callosum_fibers else 0,

        # ---------------------------------------------------------------------
        # "Segmentation" panel

        # Segmentation
        'segmentRightHemisphere':  2 if segment_right_hemisphere else 0,
        'segmentLeftHemisphere':   2 if segment_left_hemisphere else 0,
        'segmentInterHemispheric': 2 if segment_interhemispheric else 0,
        'segmentCerebellum':       2 if segment_cerebellum else 0,
        'fileNameRoiMask':         roi_mask_path if roi_mask_path else "",
        'minimumFiberLengthValue': min_fiber_length,
        'maximumFiberLengthValue': max_fiber_length,
        'segmentRois':             2 if roi_mask_path else 0,

        'labels': '01',  # Unused, for the GUI only
        'names': 'unknown',  # Unused, for the GUI only
        'roiLabelsAndNames': roi_labels_and_names,

        # ---------------------------------------------------------------------
        # "Tractography parameters" panel
        'voxelSamplerPointCount': voxel_sampler_point_count,

        # ---------------------------------------------------------------------
        # "Voxel based clustering parameters" panel
        'extractionPercent1':
        percent_of_bundle_extraction_for_first_clustering,
        'runSecondClustering': 2 if perform_a_second_clustering else 0,
        'extractionPercent2':
        percent_of_bundle_extraction_for_second_clustering,
        'allCentersMaxDistance': max_distance_for_fasicle_merging,

        # ---------------------------------------------------------------------
        # 'Advanced parameters' panel

        # "Divide fibers according to their length" subpanel
        'lengthGroup': length_groups,

        # "Parcellation" subpanel
        'minimumFiberInMask': nb_fibers_per_voxels,
        'parcelSize':         parcel_size,
        'minimumParcelSize':  min_parcel_size,

        # "Clustering" subpanel
        'minimumClusterSize':          min_cluster_size,
        'averageClusterNumber':        average_nb_of_cluster,
        'averageClusterSize':          average_cluster_size,
        'smallSizeForIsolatedCluster': small_size_for_isolated_cluster,
        'connectivityPercent':         connectivity_matrix_threshold,

        # "Dividing fibers according to their extremeties" subpanel
        'wForest': min_nb_of_fibers_by_bundle_before_watershed,
        'extremitiesThreshold': nb_of_fibers_per_voxel,
        'minimumNumberFiberInWatershed':
        min_nb_of_fibers_by_bundle_after_watershed,
        'minimumNumberFiberOutput':
        min_nb_of_fibers_by_bundle_in_output_bundle_file,

        # "Centroids" subpanel
        'centroidClosest': 1 if centroid == "closest_fiber" else 0,
        'centroidMean':    1 if centroid == "mean_fiber" else 0,

        # ---------------------------------------------------------------------
        # "Output option" panel
        'removeTemporaryFiles': remove_temporary_files,

        # ---------------------------------------------------------------------
        # "Work directory" panel
        'outputWorkDirectory': outdir,

        # ---------------------------------------------------------------------
        # Other parameters
        '_subjectName': subject_id if subject_id else "",
        'viewType':     0,  # Unused, for the GUI only
    }

    # Check exclusive arguments
    assert (parameters_dict["centroidClosest"] +
            parameters_dict["centroidMean"] == 1)
    assert (parameters_dict["subjectTypeAdult"] +
            parameters_dict["subjectTypeChild"] == 1)

    # Call with Connectomist
    connprocess = ConnectomistWrapper(path_connectomist)
    parameter_file = ConnectomistWrapper.create_parameter_file(
        algorithm, parameters_dict, outdir)
    connprocess(algorithm, parameter_file, outdir)

    return outdir
